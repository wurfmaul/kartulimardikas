for ($r[0] = 0; $r[0] < $len; $r[0]++) // loop over all elements
{
	$r[1] = $r[0]; // set the first element as current minimum
	for ($r[2] = $r[0] + 1; $r[2] < $len; $r[2]++) // loop over all remaining elements
	{
		if ($a[$r[2]] < $a[$r[1]]) // if a smaller elemnt is found...
		{
			$r[1] = $r[2]; // ...replace the current minimum 
		}
	}
	$r[3] = $a[$r[1]]; // perform exchange of first and minimum element
	$a[$r[1]] = $a[$r[0]]; //APPEND_COMMENT
	$a[$r[0]] = $r[3]; //APPEND_COMMENT
}
